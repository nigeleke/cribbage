39 object Styling:
40 
41   private def styledFace: Card.Face => String =
42     case Ace   => "A"
43     case Two   => "2"
44     case Three => "3"
45     case Four  => "4"
46     case Five  => "5"
47     case Six   => "6"
48     case Seven => "7"
49     case Eight => "8"
50     case Nine  => "9"
51     case Ten   => "T"
52     case Jack  => "J"
53     case Queen => "Q"
54     case King  => "K"
55 
56   private def styledSuit: Card.Suit => String =
57     case Hearts   => "\u2665"
58     case Clubs    => "\u2663"
59     case Diamonds => "\u2666"
60     case Spades   => "\u2660"
61 
62   private def colourOf: Card.Suit => String =
63     case Hearts | Diamonds => s"${Ansi.escape}[31m"
64     case Clubs | Spades    => s"${Ansi.escape}[30m"
65 
66   extension (card: Card)
67     def styled: String =
68       val colour = colourOf(card.suit)
69       val face   = styledFace(card.face)
70       val suit   = styledSuit(card.suit)
71       s"$colour$face$suit${Ansi.reset}"
72 
73   /** ANSI escape codes for pretty printing to console. */
74   object Ansi:
75     val escape = "\u001b"
76     val reset  = s"$escape[0m"
77 
78   extension (cards: Crib | Deck | Hand)
79     def styled: String = cards.toSeq.map(_.styled).mkString("[", " ", "]")
80 
81   extension (player: Player) def styled: String = player.id.toString.drop(30)
82 
83   extension (score: Score) def styled: String = s"${score.back}\u2192 ${score.front}"
84 
85   extension (play: Plays.Play)
86     def styled: String =
87       play match
88         case Plays.Laid(player, card) => s"${player.styled} => ${card.styled}"
89         case Plays.Pass(player)       => s"${player.styled} => pass"
90 
91   extension (plays: Plays)
92     def styled: String =
93       val inPlayStyled = plays.inPlay.map(_.styled).mkString("[", ", ", "]")
94       val playedStyled = plays.played.map(_.styled).mkString("[", ", ", "]")
95       s"nextToPlay: ${plays.nextPlayer.styled} inPlay: $inPlayStyled played: $playedStyled"
96 
97   extension (discarding: Discarding)
98     def styled: String =
99       val scoresStyled =
100         discarding.scores.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
101       val handsStyled  =
102         discarding.hands.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
103       s"""Discarding
104          |  deck:   ${discarding.deck.styled}
105          |  scores: $scoresStyled
106          |  hands:  $handsStyled
107          |  dealer: ${discarding.dealer.styled}
108          |  pone:   ${discarding.pone.styled}
109          |  crib:   ${discarding.crib.styled}""".stripMargin
110 
111   extension (playing: Playing)
112     def styled: String =
113       val scoresStyled = playing.scores.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
114       val handsStyled  = playing.hands.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
115       s"""Playing
116          |  scores: $scoresStyled
117          |  hands:  $handsStyled
118          |  dealer: ${playing.dealer.styled}
119          |  pone:   ${playing.pone.styled}
120          |  crib:   ${playing.crib.styled}
121          |  cut:    ${playing.cut.styled}
122          |  plays:  ${playing.plays.styled}""".stripMargin
123 
124   extension (finished: Finished)
125     def styled: String =
126       val scoresStyled =
127         finished.scores.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
128       s"""Finished
129          |  scores: $scoresStyled""".stripMargin
