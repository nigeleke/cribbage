36 /** Cards object manages sequence of cards behind the Crib, Deck and Hand types. */
37 object Cards:
38   opaque type Crib = Seq[Card]
39   opaque type Deck = Seq[Card]
40   opaque type Hand = Seq[Card]
41 
42   /** Create a Crib, Deck or Hand from a sequence of cards.
43     * @param cards
44     *   The cards to be included in the resulant id.
45     * @tparam T
46     *   The required Crib, Deck or Hand type.
47     * @return
48     *   An instance of T containing the Cards.
49     */
50   private def cardsOf[T <: Crib | Deck | Hand](cards: Seq[Card]): T = cards.asInstanceOf[T]
51 
52   /** Create a Crib with the given cards.
53     * @constructor
54     * @param cards
55     *   The cards.
56     * @return
57     *   The Crib.
58     */
59   def cribOf(cards: Seq[Card]): Crib = cardsOf[Crib](cards)
60 
61   /** An empty Crib. */
62   val emptyCrib: Crib = Seq.empty
63 
64   /** Create a Deck with the given cards.
65     * @constructor
66     * @param cards
67     *   The cards.
68     * @return
69     *   The Deck.
70     */
71   def deckOf(cards: Seq[Card]): Deck = cardsOf[Deck](cards)
72 
73   /** A full Deck of cards.
74     */
75   val fullDeck: Deck = (for
76     face <- Card.faces
77     suit <- Card.suits
78   yield Card(face, suit)).toIndexedSeq
79 
80   /** @return
81     *   A shuffled Deck of cards.
82     * @note
83     *   A different shuffled Deck is returned on each call.
84     */
85   def shuffledDeck: Deck = Random.shuffle(fullDeck)
86 
87   /** Create a Hand with the given cards.
88     *
89     * @constructor
90     * @param cards
91     *   The cards.
92     * @return
93     *   The Hand.
94     */
95   def handOf(cards: Seq[Card]): Hand = cardsOf[Hand](cards)
96 
97   /** An empty Hand. */
98   val emptyHand: Hand = Seq.empty
99 
100   extension (cards: Crib | Deck | Hand)
101     /** @return Return the number of cards contained in the collection. */
102     def size: Int = cards.size
103 
104     /** @return Return the cards in a non-descriptive collection. */
105     def toSeq: Seq[Card] = cards
106 
107   extension (crib: Crib)
108     /** @return True if the correct number of cards have been discarded. */
109     def isFull: Boolean = crib.size == CardsRequiredInCrib
110 
111     /** Add the provided cards to the Crib.
112       * @return
113       *   The new Crib including the new Cards.
114       */
115     def addAll(these: Seq[Card]): Crib = crib ++ these
116 
117   extension (hand: Hand)
118     /** Check if the Hand contains the provided Card.
119       *
120       * @param card
121       *   The Card.
122       * @return
123       *   True, if the Hand contains the Card, false otherwise.
124       */
125     def contains(card: Card): Boolean = hand.contains(card)
126 
127     /** Check if the Hand contains any Card the satisfies the rule predicate. */
128     def containsAny(rule: Card => Boolean): Boolean = hand.filter(rule).nonEmpty
129 
130     /** Check if the Hand contains the provided Cards.
131       *
132       * @param these
133       *   The Cards to be checked.
134       * @return
135       *   True, if the Hand contains the Card, false otherwise.
136       */
137     def containsAll(these: Seq[Card]): Boolean = these.forall(hand.contains)
138 
139     /** Remove the provided card from the Hand.
140       * @param card
141       *   The Card.
142       * @return
143       *   The updated Hand. If the Card is not present the original Hand will be returned.
144       */
145     def remove(card: Card): Hand = removeAll(Seq(card))
146 
147     /** Remove the provided cards from the Hand.
148       * @param these
149       *   The Cards to be removed.
150       * @return
151       *   The updated Hand. If the Card is not present the original Hand will be returned.
152       */
153     def removeAll(these: Seq[Card]): Hand = hand.filterNot(these.contains)
154 
155   extension (deck: Deck)
156 
157     /** Deal cards from the current Deck.
158       * @param numberOfHands
159       *   The number of hands to deal.
160       * @param cardsPerHand
161       *   The number of cards to deal in each hand.
162       * @return
163       *   A tuple of a) cards remaining from the Deck after the and b) a sequence of the Hands
164       *   dealt.
165       * @throws RuntimeException
166       *   if not enough cards in the deck.
167       */
168     def deal(numberOfHands: Int, cardsPerHand: Int): (Deck, Seq[Hand]) =
169       require(deck.size >= numberOfHands * cardsPerHand)
170       def dealHand(n: Int): Hand =
171         deck
172           .drop((numberOfHands - n) * cardsPerHand)
173           .take(cardsPerHand)
174       val hands: Seq[Hand]       = (1 to numberOfHands).map(dealHand)
175       val remainder: Deck        = deck.drop(numberOfHands * cardsPerHand)
176       (remainder, hands)
177 
178     /** Cut a card from the Deck.
179       * @return
180       *   A tuple of a) cards remaining after the cut, and b) the cut card itself.
181       * @throws RuntimeException
182       *   if not enough cards in the deck.
183       */
184     def cut: (Deck, Card) =
185       require(!deck.isEmpty)
186       val card = deck.head
187       (deck.filterNot(_ == card), card)
