36 /** The plays being made in a Playing game.
37   * @param nextPlayer
38   *   The next player to play or pass.
39   * @param inPlay
40   *   The cards in the current play.
41   * @param played
42   *   The cards previously played.
43   */
44 final case class Plays(
45     nextPlayer: Player,
46     inPlay: Seq[Plays.Play],
47     played: Seq[Plays.Play]
48 )
49 
50 object Plays:
51   /** Create a fresh plays.
52     * @param player
53     *   The first player to play.
54     * @return
55     *   The plays.
56     */
57   def apply(player: Player): Plays =
58     new Plays(nextPlayer = player, Seq.empty, Seq.empty)
59 
60   /** A single play, recording the player and their card laid or their pass.
61     */
62   sealed trait Play
63   final case class Laid(player: Player, card: Card) extends Play
64   final case class Pass(player: Player)             extends Play
65 
66   extension (plays: Plays)
67     /** Running total for the current play. Max 31. */
68     def runningTotal =
69       plays.inPlay.map { play =>
70         play match
71           case Plays.Laid(_, card) => card.value
72           case Plays.Pass(_)       => 0
73       }.sum
74 
75     def mustPass(hand: Hand) =
76       val currentTotal = plays.runningTotal
77       !hand.containsAny(_.value + currentTotal <= PlayTarget)
78 
79     /** The set of players that have Passed in the current play. */
80     private def passedPlayers = plays.inPlay
81       .collect { case p: Plays.Pass => p }
82       .map(_.player)
83       .toSet
84 
85     /** @return
86       *   The number of players that have Passed in the current play.
87       */
88     private def passCount = plays.passedPlayers.size
89 
90     def allPassed: Boolean = plays.passCount == NumberOfPlayersInGame
91 
92     def play(player: Player, card: Card): Plays =
93       val updatedInPlay = plays.inPlay :+ Plays.Laid(player, card)
94       plays.copy(inPlay = updatedInPlay)
95 
96     def pass(player: Player): Plays =
97       val updatedInPlay = plays.inPlay :+ Plays.Pass(player)
98       plays.copy(inPlay = updatedInPlay)
99 
100     def allPlaysCompleted: Boolean = allLaid.size == AllLaidCardCount
101 
102     private def allLaid: Seq[Plays.Laid] = (plays.inPlay ++ plays.played).collect {
103       case play: Plays.Laid => play
104     }
105 
106     def withNextPlayer(player: Player): Plays =
107       plays.copy(nextPlayer = player)
108 
109     def restarted: Plays =
110       val updatedPlayed = plays.played ++ plays.inPlay
111       plays.copy(inPlay = Seq.empty, played = updatedPlayed)
112 
113     /** @return
114       *   The current laid cards, without the Pass detail.
115       */
116     def cardsInPlay: Seq[Card] = plays.inPlay.collect { case Plays.Laid(_, card) => card }
117 
118     def regather: Map[Player, Hand] =
119       allLaid.groupMap(_.player)(_.card).map((p, cs) => (p, Cards.handOf(cs)))
