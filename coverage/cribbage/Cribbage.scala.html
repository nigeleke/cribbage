42 object Cribbage:
43   def newGame: Cribbage =
44     Cribbage(
45       Set(Player.newPlayer, Player.newPlayer),
46       Cards.fullDeck
47     )
48 
49   def makeDraw: Cribbage => Draw =
50     case Cribbage(players, _) => drawForPlayers(players)
51 
52   private def drawForPlayers(players: Set[Player]) =
53     val shuffled          = shuffledDeck
54     val (remaining, cut1) = shuffled.cut
55     val (_, cut2)         = remaining.cut
56     val cuts              = Seq(cut1, cut2).sortBy(_.rank)
57     val playerByCut       = cuts.zip(players).toMap
58     val cutsByPlayer      = for ((k, v) <- playerByCut) yield (v, k)
59     val sameRank          = cut1.rank == cut2.rank
60     if sameRank
61     then Draw.Undecided(cutsByPlayer)
62     else Draw.Decided(cutsByPlayer, playerByCut(cuts.head), playerByCut(cuts.last))
63 
64   def redraw: Undecided => Draw =
65     case Undecided(draws) => drawForPlayers(draws.keySet)
66 
67   def dealFirstHand: Decided => Discarding =
68     case Decided(_, dealer, pone) =>
69       val initialScores = Set(dealer, pone).map((_, Score.zero)).toMap
70       newDiscarding(initialScores, dealer, pone)
71 
72   private def newDiscarding(scores: Map[Player, Score], dealer: Player, pone: Player): Discarding =
73     val (deck, hands) = shuffledDeck.deal(NumberOfPlayersInGame, CardsDealtPerHand)
74     val dealtHands    = Set(pone, dealer).zip(hands).toMap
75     val crib          = emptyCrib
76     Discarding(deck, scores, dealtHands, dealer, pone, crib)
77 
78   def discardToCrib(player: Player, discards: Seq[Card]): Discarding => Discarding | Playing |
79     Finished =
80     case discarding @ Discarding(deck, scores, hands, dealer, pone, crib) =>
81       require(Set(dealer, pone).contains(player), NonParticipatingPlayer(player))
82       require(hands(player).containsAll(discards), UnheldCards(player, discards))
83       require(
84         hands(player).removeAll(discards).size >= CardsKeptPerHand,
85         TooManyDiscards(player, discards)
86       )
87       val updatedHand  = hands(player).removeAll(discards)
88       val updatedHands = hands.updated(player, updatedHand)
89       val updatedCrib  = crib.addAll(discards)
90       val (_, cut)     = deck.cut
91       if updatedCrib.isFull
92       then scoreHisHeels(Playing(scores, updatedHands, dealer, pone, updatedCrib, cut, Plays(pone)))
93       else discarding.copy(hands = updatedHands, crib = updatedCrib)

95   def play(player: Player, card: Card): Playing => Playing | Discarding | Finished =
96     case playing @ Playing(_, hands, dealer, pone, _, _, plays) =>
97       def opponent(player: Player): Player = if player == dealer then pone else dealer
98 
99       require(Set(dealer, pone).contains(player), NonParticipatingPlayer(player))
100       require(plays.nextPlayer == player, NotYourTurn(player))
101       require(hands(player).contains(card), UnheldCard(player, card))
102       require(plays.runningTotal + card.value <= PlayTarget, PlayTargetExceeded(player, card))
103 
104       val updatedHand  = hands(player).remove(card)
105       val updatedHands = hands.updated(player, updatedHand)
106       val updatedPlays = plays.play(player, card)
107       scorePlay(playing.copy(hands = updatedHands, plays = updatedPlays)) match
108         case scoredPlay: Playing =>
109           if updatedPlays.allPlaysCompleted
110           then
111             val regatheredHands = updatedPlays.regather
112             val resetPlays      = Plays(pone)
113             scoreHands(scoredPlay.copy(hands = regatheredHands, plays = resetPlays)) match
114               case scoredHands: Playing => newDiscarding(scoredHands.scores, pone, dealer)
115               case finished: Finished   => finished
116           else scoredPlay.copy(plays = updatedPlays.withNextPlayer(opponent(player)))
117         case finished: Finished  => finished
118 
119   def pass(player: Player): Playing => Playing | Finished =
120     case playing @ Playing(_, hands, dealer, pone, _, _, plays) =>
121       def opponent(player: Player): Player = if player == dealer then pone else dealer
122 
123       require(Set(dealer, pone).contains(player), NonParticipatingPlayer(player))
124       require(plays.nextPlayer == player, NotYourTurn(player))
125       require(plays.mustPass(hands(player)), PlayTargetAttainable(player))
126 
127       val updatedPlays = plays.pass(player)
128 
129       scorePass(playing.copy(plays = updatedPlays)) match
130         case scoredPlaying: Playing =>
131           if updatedPlays.allPassed
132           then scoredPlaying.copy(plays = updatedPlays.restarted)
133           else scoredPlaying.copy(plays = updatedPlays.withNextPlayer(opponent(player)))
134         case finished: Finished     => finished
