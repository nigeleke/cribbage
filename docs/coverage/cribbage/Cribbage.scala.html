42 object Cribbage:
43   def newGame: Cribbage[Draw] = drawForPlayers(Set(Player.newPlayer, Player.newPlayer))

45   private def drawForPlayers(players: Set[Player]) =
46     val shuffled          = shuffledDeck
47     val (remaining, cut1) = shuffled.cut
48     val (_, cut2)         = remaining.cut
49     val cuts              = Seq(cut1, cut2).sortBy(_.rank)
50     val playerByCut       = cuts.zip(players).toMap
51     val cutsByPlayer      = for ((k, v) <- playerByCut) yield (v, k)
52     val sameRank          = cut1.rank == cut2.rank
53     if sameRank
54     then Draw.Undecided(cutsByPlayer)
55     else Draw.Decided(cutsByPlayer, playerByCut(cuts.head), playerByCut(cuts.last))
56 
57   def redraw: Undecided => Draw =
58     case Undecided(draws) => drawForPlayers(draws.keySet)
59 
60   def dealFirstHand: Decided => Discarding =
61     case Decided(_, dealer, pone) =>
62       val initialScores = Set(dealer, pone).map((_, Score.zero)).toMap
63       newDiscarding(initialScores, dealer, pone)

65   private def newDiscarding(scores: Map[Player, Score], dealer: Player, pone: Player): Discarding =
66     val (deck, hands) = shuffledDeck.deal(NumberOfPlayersInGame, CardsDealtPerHand)
67     val dealtHands    = Set(pone, dealer).zip(hands).toMap
68     val crib          = emptyCrib
69     Discarding(deck, scores, dealtHands, dealer, pone, crib)

71   def discardToCrib(player: Player, discards: Seq[Card]): Discarding => Discarding | Playing |
72     Finished =
73     case discarding @ Discarding(deck, scores, hands, dealer, pone, crib) =>
74       require(Set(dealer, pone).contains(player), NonParticipatingPlayer(player))
75       require(hands(player).containsAll(discards), UnheldCards(player, discards))
76       require(
77         hands(player).removeAll(discards).size >= CardsKeptPerHand,
78         TooManyDiscards(player, discards)
79       )
80       val updatedHand  = hands(player).removeAll(discards)
81       val updatedHands = hands.updated(player, updatedHand)
82       val updatedCrib  = crib.addAll(discards)
83       val (_, cut)     = deck.cut
84       if updatedCrib.isFull
85       then scoreHisHeels(Playing(scores, updatedHands, dealer, pone, updatedCrib, cut, Plays(pone)))
86       else discarding.copy(hands = updatedHands, crib = updatedCrib)

88   def play(player: Player, card: Card): Playing => Playing | Discarding | Finished =
89     case playing @ Playing(_, hands, dealer, pone, _, _, plays) =>
90       def opponent(player: Player): Player = if player == dealer then pone else dealer
91 
92       require(Set(dealer, pone).contains(player), NonParticipatingPlayer(player))
93       require(plays.nextPlayer == player, NotYourTurn(player))
94       require(hands(player).contains(card), UnheldCard(player, card))
95       require(plays.runningTotal + card.value <= PlayTarget, PlayTargetExceeded(player, card))
96 
97       val updatedHand  = hands(player).remove(card)
98       val updatedHands = hands.updated(player, updatedHand)
99       val updatedPlays = plays.play(player, card)
100       scorePlay(playing.copy(hands = updatedHands, plays = updatedPlays)) match
101         case scoredPlay: Playing =>
102           if updatedPlays.allPlaysCompleted
103           then
104             val regatheredHands = updatedPlays.regather
105             val resetPlays      = Plays(pone)
106             scoreHands(scoredPlay.copy(hands = regatheredHands, plays = resetPlays)) match
107               case scoredHands: Playing => newDiscarding(scoredHands.scores, pone, dealer)
108               case finished: Finished   => finished
109           else scoredPlay.copy(plays = updatedPlays.withNextPlayer(opponent(player)))
110         case finished: Finished  => finished
111 
112   def pass(player: Player): Playing => Playing | Finished =
113     case playing @ Playing(_, hands, dealer, pone, _, _, plays) =>
114       def opponent(player: Player): Player = if player == dealer then pone else dealer
115 
116       require(Set(dealer, pone).contains(player), NonParticipatingPlayer(player))
117       require(plays.nextPlayer == player, NotYourTurn(player))
118       require(plays.mustPass(hands(player)), PlayTargetAttainable(player))
119 
120       val updatedPlays = plays.pass(player)
121 
122       scorePass(playing.copy(plays = updatedPlays)) match
123         case scoredPlaying: Playing =>
124           if updatedPlays.allPassed
125           then scoredPlaying.copy(plays = updatedPlays.restarted)
126           else scoredPlaying.copy(plays = updatedPlays.withNextPlayer(opponent(player)))
127         case finished: Finished     => finished
