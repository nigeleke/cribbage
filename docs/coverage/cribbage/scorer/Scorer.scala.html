37 /** The Scorer is used to score different stages of the Cribbage game:
38   *   - His Heels (when card is cut after discarding and prior to playing,
39   *   - During Play (including when a Player has passed)
40   *   - After all Plays, when pone / dealer hands and crib are scored.
41   *
42   * All scores may result in a Finished game if the points scored make the players total score
43   * greater than 121.
44   */
45 object Scorer:
46 
47   /** Score his heels - if a Jack is cut prior to Play. */
48   val scoreHisHeels: Playing => Playing | Finished =
49     case playing @ Playing(currentScores, _, dealer, _, _, cut, _) =>
50       if cut.face == Card.Face.Jack
51       then scoreFor(dealer, 2)(playing)
52       else playing
53 
54   private def scoreFor(player: Player, points: Int): Playing => Playing | Finished =
55     case playing @ Playing(scores, _, _, _, _, _, _) =>
56       val updatedScore  = scores(player).add(points)
57       val updatedScores = scores.updated(player, updatedScore)
58       if updatedScore.isWinningScore
59       then Finished(scores = updatedScores)
60       else playing.copy(scores = updatedScores)
61 
62   /** Score the current play as a result of the last card laid. This includes fifteens, runs, pairs
63     * and final card play.
64     */
65   val scorePlay: Playing => Playing | Finished =
66     case playing @ Playing(scores, hands, dealer, pone, crib, cut, plays) =>
67       val fifteens   = if plays.runningTotal == 15 then 2 else 0
68       val pairs      =
69         val reversed      = plays.cardsInPlay.reverse
70         val optPlayedCard = reversed.headOption
71         (for {
72           lastCard     <- optPlayedCard
73           matchingCards = reversed.drop(1).takeWhile(_.face == lastCard.face)
74           size          = matchingCards.size
75           points        = Map(0 -> 0, 1 -> 2, 2 -> 6, 3 -> 12)(size)
76         } yield points).getOrElse(0)
77       val runs       =
78         val reversed      = plays.cardsInPlay.reverse
79         val optPlayedCard = reversed.headOption
80         val runLengths    = reversed.size to 3 by -1
81         (for
82           playedCard    <- optPlayedCard
83           bestRunLength <-
84             runLengths.dropWhile(length => !makesRun(playedCard, reversed.take(length))).headOption
85         yield bestRunLength).getOrElse(0)
86       val play       = (plays.runningTotal, plays.allPlaysCompleted) match
87         case (PlayTarget, _) => 2
88         case (_, true)       => 1
89         case (_, false)      => 0
90       val totalScore = fifteens + pairs + runs + play
91       scoreFor(plays.nextPlayer, totalScore)(playing)
92 
93   private def isRun(cards: Seq[Card]) =
94     val sorted              = cards.sortBy(_.rank)
95     val differences         = sorted.sliding(2).map { case Seq(x, y, _*) => y.rank - x.rank }
96     val differencesNotByOne = differences.filterNot(_ == 1)
97     differencesNotByOne.isEmpty
98 
99   private def makesRun(playedCard: Card, cards: Seq[Card]) =
100     isRun(cards) && cards.contains(playedCard)
101 
102   /** Scores one for the final play, if neither player acn lay any further cards and the target 31
103     * wasn't reached.
104     */
105   val scorePass: Playing => Playing | Finished =
106     case playing @ Playing(_, _, _, _, _, _, plays) =>
107       val passScore = if plays.allPassed then 1 else 0
108       scoreFor(plays.nextPlayer, passScore)(playing) // 2 for 31 scored in scorePlay

110   /** Scores the pone's hand, dealer's hand then crib. Any score may result in a finished game, in
111     * which case subsequent hands aren't scored.
112     */
113   val scoreHands: Playing => Playing | Finished =
114     case playing: Playing => (scorePoneHand andThen scoreDealerHand andThen scoreCrib)(playing)
115 
116   private val scorePoneHand: Playing => Playing | Finished =
117     case playing @ Playing(_, hands, _, pone, _, cut, _) =>
118       scoreCards(pone, hands(pone).toSeq, cut, false)(playing)
119 
120   private val scoreDealerHand: Playing | Finished => Playing | Finished =
121     case playing @ Playing(_, hands, dealer, _, _, cut, _) =>
122       scoreCards(dealer, hands(dealer).toSeq, cut, false)(playing)
123     case finished: Finished                                => finished
124 
125   private val scoreCrib: Playing | Finished => Playing | Finished =
126     case playing @ Playing(_, _, dealer, _, crib, cut, _) =>
127       scoreCards(dealer, crib.toSeq, cut, true)(playing)
128     case finished: Finished                               => finished
129 
130   private def scoreCards(
131       player: Player,
132       cards: Seq[Card],
133       cut: Card,
134       inCrib: Boolean
135   ): Playing => Playing | Finished =
136     case playing: Playing =>
137       val allCards = cards.toSeq :+ cut
138 
139       val fifteens =
140         val nCards    = 2 to 5
141         val nFifteens = for {
142           n    <- nCards
143           c    <- allCards.combinations(n)
144           total = c.map(_.value).sum if total == 15
145         } yield ("fifteen: ", c)
146         nFifteens.size * 2
147 
148       val pairs =
149         val nPairs = for {
150           c     <- allCards.combinations(2)
151           c1    <- c.headOption
152           c2    <- c.lastOption
153           isPair = c1.face == c2.face if isPair
154         } yield ("pair: ", c)
155         nPairs.size * 2
156 
157       val runs =
158         val nCards  = 3 to 5
159         val allRuns = (for {
160           n <- nCards
161           c <- allCards.combinations(n) if isRun(c)
162         } yield c).groupBy(_.size)
163 
164         val (count, length) =
165           if (allRuns.isEmpty) (0, 0)
166           else {
167             val max = allRuns.keySet.max
168             (allRuns(max).size, max)
169           }
170 
171         count * length
172 
173       val heels = (for {
174         card <- cards.toSeq if card.face == Card.Face.Jack && card.suit == cut.suit
175       } yield card).length
176 
177       val flushes =
178         val allFlush   = allCards.size == 5 && allCards.groupBy(_.suit).size == 1
179         val cardsFlush = !inCrib && cards.size == 4 && cards.toSeq.groupBy(_.suit).size == 1
180         if (allFlush) 5
181         else if (cardsFlush) 4
182         else 0
183 
184       scoreFor(player, pairs + fifteens + runs + flushes + heels)(playing)
