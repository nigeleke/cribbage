34 /** A playing card.
35   *
36   * @param face
37   *   The card's face.
38   * @param suit
39   *   The card's suit.
40   */
41 final case class Card(face: Card.Face, suit: Card.Suit)
42 
43 object Card:
44   /** A Card.Face, providing the face id, the run & cut ranking, and a short string for printing.
45     */
46   enum Face(val value: Int, val rank: Int):
47     case Ace   extends Face(1, 1)
48     case Two   extends Face(2, 2)
49     case Three extends Face(3, 3)
50     case Four  extends Face(4, 4)
51     case Five  extends Face(5, 5)
52     case Six   extends Face(6, 6)
53     case Seven extends Face(7, 7)
54     case Eight extends Face(8, 8)
55     case Nine  extends Face(9, 9)
56     case Ten   extends Face(10, 10)
57     case Jack  extends Face(10, 11)
58     case Queen extends Face(10, 12)
59     case King  extends Face(10, 13)
60 
61   /** A Card.Suit providing a symbolic string and suit colour for printing.
62     */
63   enum Suit:
64     case Clubs, Diamonds, Hearts, Spades
65 
66   /** Expose the faces as a collection. */
67   val faces = Face.values
68 
69   /** Expose the suits as a collection. */
70   val suits = Suit.values
71 
72   extension (card: Card)
73     /** Expose the card's rank. */
74     def rank = card.face.rank
75 
76     /** Expose the card's face id. */
77     def value = card.face.value
