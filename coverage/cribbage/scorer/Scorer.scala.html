39   val scoreHisHeels: Playing => Playing | Finished =
40     case playing @ Playing(currentScores, _, dealer, _, _, cut, _) =>
41       if cut.face == Card.Face.Jack
42       then scoreFor(dealer, 2)(playing)
43       else playing

45   private def scoreFor(player: Player, points: Int): Playing => Playing | Finished =
46     case playing @ Playing(scores, _, _, _, _, _, _) =>
47       val updatedScore  = scores(player).add(points)
48       val updatedScores = scores.updated(player, updatedScore)
49       if updatedScore.isWinningScore
50       then Finished(scores = updatedScores)
51       else playing.copy(scores = updatedScores)
52 
53   val scorePlay: Playing => Playing | Finished =
54     case playing @ Playing(scores, hands, dealer, pone, crib, cut, plays) =>
55       val fifteens   = if plays.runningTotal == 15 then 2 else 0
56       val pairs      =
57         val reversed      = plays.cardsInPlay.reverse
58         val optPlayedCard = reversed.headOption
59         (for {
60           lastCard     <- optPlayedCard
61           matchingCards = reversed.drop(1).takeWhile(_.face == lastCard.face)
62           size          = matchingCards.size
63           points        = Map(0 -> 0, 1 -> 2, 2 -> 6, 3 -> 12)(size)
64         } yield points).getOrElse(0)
65       val runs       =
66         val reversed      = plays.cardsInPlay.reverse
67         val optPlayedCard = reversed.headOption
68         val runLengths    = reversed.size to 3 by -1
69         (for
70           playedCard    <- optPlayedCard
71           bestRunLength <-
72             runLengths.dropWhile(length => !makesRun(playedCard, reversed.take(length))).headOption
73         yield bestRunLength).getOrElse(0)
74       val play       = (plays.runningTotal, plays.allPlaysCompleted) match
75         case (PlayTarget, _) => 2
76         case (_, true)       => 1
77         case (_, false)      => 0
78       val totalScore = fifteens + pairs + runs + play
79       scoreFor(plays.nextPlayer, totalScore)(playing)

81   private def isRun(cards: Seq[Card]) =
82     val sorted              = cards.sortBy(_.rank)
83     val differences         = sorted.sliding(2).map { case Seq(x, y, _*) => y.rank - x.rank }
84     val differencesNotByOne = differences.filterNot(_ == 1)
85     differencesNotByOne.isEmpty
86 
87   private def makesRun(playedCard: Card, cards: Seq[Card]) =
88     isRun(cards) && cards.contains(playedCard)

112   private def scoreCards(
113       player: Player,
114       cards: Seq[Card],
115       cut: Card,
116       inCrib: Boolean
117   ): Playing => Playing | Finished =
118     case playing: Playing =>
119       val allCards = cards.toSeq :+ cut
120 
121       val fifteens =
122         val nCards    = 2 to 5
123         val nFifteens = for {
124           n    <- nCards
125           c    <- allCards.combinations(n)
126           total = c.map(_.value).sum if total == 15
127         } yield ("fifteen: ", c)
128         nFifteens.size * 2
129 
130       val pairs =
131         val nPairs = for {
132           c     <- allCards.combinations(2)
133           c1    <- c.headOption
134           c2    <- c.lastOption
135           isPair = c1.face == c2.face if isPair
136         } yield ("pair: ", c)
137         nPairs.size * 2
138 
139       val runs =
140         val nCards  = 3 to 5
141         val allRuns = (for {
142           n <- nCards
143           c <- allCards.combinations(n) if isRun(c)
144         } yield c).groupBy(_.size)
145 
146         val (count, length) =
147           if (allRuns.isEmpty) (0, 0)
148           else {
149             val max = allRuns.keySet.max
150             (allRuns(max).size, max)
151           }
152 
153         count * length
154 
155       val heels = (for {
156         card <- cards.toSeq if card.face == Card.Face.Jack && card.suit == cut.suit
157       } yield card).length
158 
159       val flushes =
160         val allFlush   = allCards.size == 5 && allCards.groupBy(_.suit).size == 1
161         val cardsFlush = !inCrib && cards.size == 4 && cards.toSeq.groupBy(_.suit).size == 1
162         if (allFlush) 5
163         else if (cardsFlush) 4
164         else 0
165 
166       scoreFor(player, pairs + fifteens + runs + flushes + heels)(playing)
