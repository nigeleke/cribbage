39 /** Provides styling for pretty-printing of the model objects to an ANSI terminal.
40   */
41 object Styling:
42 
43   private def styledFace: Card.Face => String =
44     case Ace   => "A"
45     case Two   => "2"
46     case Three => "3"
47     case Four  => "4"
48     case Five  => "5"
49     case Six   => "6"
50     case Seven => "7"
51     case Eight => "8"
52     case Nine  => "9"
53     case Ten   => "T"
54     case Jack  => "J"
55     case Queen => "Q"
56     case King  => "K"
57 
58   private def styledSuit: Card.Suit => String =
59     case Hearts   => "\u2665"
60     case Clubs    => "\u2663"
61     case Diamonds => "\u2666"
62     case Spades   => "\u2660"
63 
64   private def colourOf: Card.Suit => String =
65     case Hearts | Diamonds => s"${Ansi.escape}[31m"
66     case Clubs | Spades    => s"${Ansi.escape}[30m"
67 
68   extension (card: Card)
69     def styled: String =
70       val colour = colourOf(card.suit)
71       val face   = styledFace(card.face)
72       val suit   = styledSuit(card.suit)
73       s"$colour$face$suit${Ansi.reset}"
74 
75   /** ANSI escape codes for pretty printing to console. */
76   private object Ansi:
77     val escape = "\u001b"
78     val reset  = s"$escape[0m"
79 
80   extension (cards: Crib | Deck | Hand)
81     def styled: String = cards.toSeq.map(_.styled).mkString("[", " ", "]")
82 
83   extension (player: Player) def styled: String = player.id.toString.drop(30)
84 
85   extension (score: Score) def styled: String = s"${score.back}\u2192 ${score.front}"
86 
87   extension (play: Plays.Play)
88     def styled: String =
89       play match
90         case Plays.Laid(player, card) => s"${player.styled} => ${card.styled}"
91         case Plays.Pass(player)       => s"${player.styled} => pass"
92 
93   extension (plays: Plays)
94     def styled: String =
95       val inPlayStyled = plays.inPlay.map(_.styled).mkString("[", ", ", "]")
96       val playedStyled = plays.played.map(_.styled).mkString("[", ", ", "]")
97       s"nextToPlay: ${plays.nextPlayer.styled} inPlay: $inPlayStyled played: $playedStyled"
98 
99   extension (discarding: Discarding)
100     def styled: String =
101       val scoresStyled =
102         discarding.scores.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
103       val handsStyled  =
104         discarding.hands.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
105       s"""Discarding
106          |  deck:   ${discarding.deck.styled}
107          |  scores: $scoresStyled
108          |  hands:  $handsStyled
109          |  dealer: ${discarding.dealer.styled}
110          |  pone:   ${discarding.pone.styled}
111          |  crib:   ${discarding.crib.styled}""".stripMargin
112 
113   extension (playing: Playing)
114     def styled: String =
115       val scoresStyled = playing.scores.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
116       val handsStyled  = playing.hands.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
117       s"""Playing
118          |  scores: $scoresStyled
119          |  hands:  $handsStyled
120          |  dealer: ${playing.dealer.styled}
121          |  pone:   ${playing.pone.styled}
122          |  crib:   ${playing.crib.styled}
123          |  cut:    ${playing.cut.styled}
124          |  plays:  ${playing.plays.styled}""".stripMargin
125 
126   extension (finished: Finished)
127     def styled: String =
128       val scoresStyled =
129         finished.scores.map((k, v) => s"${k.styled} => ${v.styled}").mkString(" / ")
130       s"""Finished
131          |  scores: $scoresStyled""".stripMargin
